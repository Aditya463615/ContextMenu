window.addEventListener('contextmenu', function (e) { e.preventDefault();
const contextBox = document.getElementById('contextMenu');
const pageWidth = parseInt(window.innerWidth);
const pageHeight = parseInt(window.innerHeight);
const contextWidth = parseInt(document.getElementById('contextMenu').offsetWidth);
const contextHeight = parseInt(document.getElementById('contextMenu').offsetHeight);
var X = event.clientX;
var Y = event.clientY;
var Xoff = pageWidth-X;
var Yoff = pageHeight-Y;
//alert(Xoff+'\n'+Yoff);
if (Xoff > contextWidth && Yoff > contextHeight){
  contextBox.style.top = Y+'px';
  contextBox.style.left = X+'px';
  contextBox.style.right = 'auto';
  contextBox.style.bottom = 'auto';
  contextBox.style.transform = 'scale(1)'
  contextBox.style.transformOrigin = 'top left';
} else if (Xoff <= contextWidth && Yoff > contextHeight) {
  contextBox.style.top = Y+'px';
  contextBox.style.right = '2px';
  contextBox.style.left = 'auto';
  contextBox.style.bottom = 'auto';
  contextBox.style.transform = 'scale(1)'
  contextBox.style.transformOrigin = 'top right';
} else if (Xoff > contextWidth && Yoff <= contextHeight) {
  contextBox.style.bottom = Yoff+'px';
  contextBox.style.left = X+'px';
  contextBox.style.right = 'auto';
  contextBox.style.top = 'auto';
  contextBox.style.transform = 'scale(1)'
  contextBox.style.transformOrigin = 'bottom left';
} else if (Xoff <= contextWidth && Yoff <= contextHeight) {
  contextBox.style.bottom = Yoff+'px';
  contextBox.style.right = '2px';
  contextBox.style.left = 'auto';
  contextBox.style.top = 'auto';
  contextBox.style.transform = 'scale(1)'
  contextBox.style.transformOrigin = 'bottom right';
}
 }); window.addEventListener('click', function (e) { document.getElementById('contextMenu').style.transform = 'scale(0)'; });
